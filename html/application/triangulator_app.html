 21def check_valid_uuid(uuid_str: str) -> None:
 22    """Fonction utilitaire. Vérifie si la chaîne fournie est un UUID valide.
 23
 24    Cette fonctios est utilisée par la couche serveur du 
 25    Triangulateur (triangulation_get) pour intercepeter les erreurs de format liés
 26    au uuid envoyé par le Client (400 Bad Request) avant d'appeler le pipeline logique.
 27
 28    Args:
 29        uuid_str: Le uuid reçu par le Client.
 30
 31    Returns:
 32        None : Si le uuid, la fonction ne retourne rien
 33
 34    Raises:
 35        ValueError: Levée si la chaîne n'a pas le format UUID standard.
 36    
 37    """
 38    if not isinstance(uuid_str, str):
 39        raise ValueError("L'argument doit être une chaîne de caractères")
 40
 41    try:
 42        UUID(uuid_str)
 43    except ValueError as err:
 44        raise ValueError(
 45            f"""le UUID fournit n'est pas au format standard 
 46            (ex: xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx) :
 47            UUID reçu : {uuid_str}"""
 48        ) from err

 50def validate_point_set(pointSet: bytes) -> None:
 51    """Fonction utilitaire. Vérifie l'intégrite d'un point set au format binaire.
 52
 53    Elle retourne None si le PointSet est valide, sinon elle génère une erreur.
 54    
 55    Args :
 56        - PointSet ------------------------------------------------
 57            Binary representation of a 2D point set.
 58            First 4 bytes (unsigned long): Number of points (N).
 59            Following N * 8 bytes: The points, where each point is:
 60            4 bytes (float): X coordinate
 61            4 bytes (float): Y coordinate
 62        -----------------------------------------------------------    
 63
 64    Returns :
 65        - None : Ne retourne rien si le PointSet est intègre
 66
 67    Raises :
 68        - Value Error : Si le PointSet n'est pas cohérent 
 69        (ex. On annonce 10 points, mais sa longueuer en stocke un nombre différent)
 70    """
 71    HEADER_SIZE = 4
 72    POINT_SIZE = 8
 73    MAX_REASONABLE_POINTS = 100000
 74
 75    if not isinstance (pointSet, bytes):
 76        raise ValueError("""Not-valid PointSet - Input must be bytes""")
 77    
 78    if len(pointSet) < HEADER_SIZE:
 79        raise ValueError(f"""Le PointSet est trop court.
 80                         Longeur minimale : {HEADER_SIZE}
 81                         Longuer du PointSet : {len(pointSet)}.""")
 82    
 83    (N,) = struct.unpack("I", pointSet[:4])
 84
 85    if N >= MAX_REASONABLE_POINTS:
 86        raise ValueError("Warning - Number of points close to maximum storageble amount")
 87
 88
 89    EXPECTED_LENGTH = HEADER_SIZE + N * POINT_SIZE
 90       
 91    if len(pointSet) != EXPECTED_LENGTH:
 92        raise ValueError("Not-valid PointSet - Point amount mismatch Announced amount")
 93    
 94    coords = struct.unpack("f" * (N*2), pointSet[HEADER_SIZE:])
 95
 96    is_nan = any(x != x for x in coords)
 97
 98    is_inf = any(abs(x) == float('inf') for x in coords)
 99
100    if is_nan or is_inf:
101        raise ValueError("""Not-valid PointSet - Corrupted content 
102                         (NaN, +- inf : no mathematical meaning""")
103    return None

139    - le traitement logique (triangulation),
140    - les vérifications post-traitement,
141
142    Args :
143         pointSetId : identifiant du PS, string au format UUID
144
145    Returns :
146         triangles : Triangles en format binaire (correspon au code 200 OK)
147

149           400 Bad Request : PointSetId n'a pas le bon format (UUID)

175        triangulation_result = triangulation_pipeline(point_set_binary)
176    except RuntimeError as e:
177        raise InternalServerError("Triangulation Failed") from e
178    
179    # -----BLOCK COMMUNICATION--------
180    # - return triangulation ps as binary
181    return triangulation_result

222def decode_binary_point_set_to_geometric(pointSet: bytes, 
223                                         tol: float | None = None) -> PointSet_Geom:
224    """Décode le PointSet binaire en représentation géométrique pour le trianguler.
225    
226    Cette fonction convertit le flux binaire brut en une liste de points
227    (coordonnées X, Y).
228
229    Args:
230        pointSet (bytes) : Le flux d'octets du PointSet récupéré du PSM.
231        tol : allow for setting a tolerance value to ignore small float oscillations 
232    Returns:
233        PointSet_Geom : La structure utilisable par le calcul
234        (ex: liste de tuples (x, y)).
235    
236    """
237    HEADER_SIZE = 4
238
239    (N,) = struct.unpack("I", pointSet[:HEADER_SIZE])
240
241    coords = struct.unpack("f" * (N * 2), pointSet[HEADER_SIZE:])
242
243    point_set_geom: PointSet_Geom = []
244    for i in range(N):
245        x, y = coords[i*2], coords[i*2+1]
246        if tol is not None:
247            # Round to nearest multiple of tol
248            x = round(x / tol) * tol
249            y = round(y / tol) * tol
250        point_set_geom.append({"x": x, "y": y})
251
252    return point_set_geom

297def are_points_collinear(point1: PointGeom, point2: PointGeom, point3: PointGeom) -> bool:
298    """Vérifie si trois points ne sont pas colinéaires.
299    
300    Args:
301        point1: Premier point
302        point2: Deuxième point  
303        point3: Troisième point
304
305    Returns:
306        not_collinear : Vrai si les trois points sont non colinéaires, Faux sinon
307
308    """
309    collinear = False
310
311    area2 = (
312        (point2["x"] - point1["x"]) * (point3["y"] - point1["y"]) -
313        (point2["y"] - point1["y"]) * (point3["x"] - point1["x"])
314    )
315
316    if (abs(area2)) < 1e-9:
317        collinear = True
318    
319    return collinear

321def two_points_distance(point1: PointGeom, point2: PointGeom) -> float:
322    """Retourne la valeur absolue de la distance euclidienne entre deux points.
323    
324    Args:
325        point1: Premier point
326        point2: Deuxième point
327        
328    Returns:
329        float: Distance entre les deux points
330    
331    """
332    dx = point2["x"] - point1["x"]
333    dy = point2["y"] - point1["y"]
334    
335    return (dx**2 + dy**2)**0.5

337def point_in_triangle(px, py, triangle, points):
338    """Check if a point (px, py) lies inside or on the boundary of a 2D triangle.
339
340    This function uses barycentric coordinates to determine if the test point
341    is within the triangular region defined by the vertices.
342
343    Args:
344        px (float): The X-coordinate of the point to check.
345        py (float): The Y-coordinate of the point to check.
346        triangle (dict): A dictionary representing the triangle, typically
347            containing vertex indices {'v1', 'v2', 'v3'}.
348        points (list[dict]): A list where each element is a dictionary
349            containing the coordinates of the vertices, e.g., [{'x': 1.0, 'y': 2.0}, ...].
350
351    Returns:
352        bool: True if the point (px, py) is inside or on the boundary of the
353            triangle, False otherwise.
354    
355    """
356    v1, v2, v3 = triangle['v1'], triangle['v2'], triangle['v3']
357    ax, ay = points[v1]['x'], points[v1]['y']
358    bx, by = points[v2]['x'], points[v2]['y']
359    cx, cy = points[v3]['x'], points[v3]['y']
360    
361    # Barycentric coordinates
362    denom = ((by - cy)*(ax - cx) + (cx - bx)*(ay - cy))
363    
364    alpha = ((by - cy)*(px - cx) + (cx - bx)*(py - cy)) / denom
365    beta = ((cy - ay)*(px - cx) + (ax - cx)*(py - cy)) / denom
366    gamma = 1.0 - alpha - beta
367    
368    return (0 <= alpha <= 1) and (0 <= beta <= 1) and (0 <= gamma <= 1)

371def edge_crosses_triangles(p_idx, q_idx, triangles_geom, points):
372    """Check if a potential edge (p, q) would cross any existing trianglesthat share the starting vertex p.
373
374    This is a local anti-crossing check: an edge (p, q) is invalid if a
375    point infinitesimally close to p, moving toward q, falls inside an
376    existing triangle containing p.
377
378    Args:
379        p_idx (int): Index of the start point p.
380        q_idx (int): Index of the end point q.
381        triangles_geom (list[dict]): List of existing triangles (vertex indices).
382        points (list[dict]): List of all point coordinates.
383
384    Returns:
385        bool: True if the edge extension crosses an existing triangle boundary, 
386            False otherwise. 
387
388    """
389    # Find triangles containing p
390    triangles_with_p = [
391        t for t in triangles_geom 
392        if p_idx in (t['v1'], t['v2'], t['v3'])
393    ]
394    
395    if not triangles_with_p:
396        return False
397    
398    # Get coordinates
399    px, py = points[p_idx]['x'], points[p_idx]['y']
400    qx, qy = points[q_idx]['x'], points[q_idx]['y']
401    
402    # Vector from p to q
403    dx = qx - px
404    dy = qy - py
405    
406    # Distance
407    dist_sq = dx*dx + dy*dy
408    
409    # Create point very close to p in direction of q
410    epsilon = 1e-5
411    # Normalize and scale
412    scale = epsilon / (dist_sq**0.5)
413    p_prime_x = px + dx * scale
414    p_prime_y = py + dy * scale
415    
416    # Check if p' is inside any triangle containing p
417    return any(point_in_triangle(p_prime_x, p_prime_y, triangle, points) 
418            for triangle in triangles_with_p)

421def triangulation_compute(pointSet_geom: PointSet_Geom) -> Triangulation_Result:
422    """Réalise l'algorithme de triangulation pour un ensemble de points en entrée.
423
424    Implémente un algorithme glouton (greedy) et déterministe qui garantit
425    qu'un même ensemble de points produira toujours la même triangulation.
426    
427    L'algorithme procède de la manière suivante :
428    1. Pour chaque point, il identifie ses deux plus proches voisins
429    2. Il crée des triangles en connectant chaque point avec ses deux voisins
430    3. Il continue le processus tant qu'il reste des points non connectés
431    4. Il vérifie et évite les intersections entre triangles
432    
433    Note: Cet algorithme ne produit pas toutes les triangulations possibles,
434    mais une triangulation valide et cohérente.
435
436    Args:
437        pointSet_geom : L'ensemble des points dont on souhaite réaliser 
438          la triangulation, au format géométrique (PointSet_Geom).
439
440    Returns:
441        triangles_geom : L'ensemble des triangles produits par la triangulation,
442          au format géométrique (Triangles_Geom).
443
444    Raises:
445        - 500 Internal Server Error : L'algorithme de triangulation échoue
446
447    """
448    # Verify if enough points
449    if len(pointSet_geom) == 0:  # Empty
450        raise Exception("Le PointSet est vide")
451    elif len(pointSet_geom) < 3:  # Not enough point
452        raise Exception("Pas assez de points : le PointSet doit contenir au moins trois points")
453
454    # Check if all points are different
455    if not all_points_are_different(pointSet_geom):
456        raise Exception("Point Set non valide : au moins deux points sont identiques : tous doivent être différents")
457
458    # Check if all points are not collinear
459    if are_all_points_collinear(pointSet_geom):
460        raise Exception("Tous les points sont alignés.")
461
462    original_points = pointSet_geom.copy()
463    indexed_points = list(enumerate(original_points))
464    indexed_points.sort(key=lambda item: (item[1]['x'], item[1]['y']))
465
466    n = len(original_points)
467    triangles_geom: Triangles_Geom = []
468    
469    used_in_triangle = [False] * n
470    adjacency_matrix = [[False] * n for _ in range(n)]
471
472    def edge_exists(i, j):
473        return adjacency_matrix[i][j] or adjacency_matrix[j][i]
474
475    def add_edge(i, j):
476        adjacency_matrix[i][j] = True
477        adjacency_matrix[j][i] = True
478
479    # Process each point
480    for current_idx in range(n):
481        orig_idx_i, point_i = indexed_points[current_idx]
482
483        if used_in_triangle[orig_idx_i]:
484            continue
485        
486        distances = []
487        for other_idx in range(n):
488            if other_idx == current_idx:
489                continue
490            
491            orig_idx_j, point_j = indexed_points[other_idx]
492            dist = two_points_distance(point_i, point_j)
493            distances.append((dist, other_idx, orig_idx_j, point_j))
494
495        distances.sort(key=lambda x: x[0])
496                
497        # Try the two closest points
498        closest1 = distances[0]
499        closest2 = distances[1]
500
501        orig_idx1 = closest1[2]
502        orig_idx2 = closest2[2]
503        point1 = closest1[3]
504        point2 = closest2[3]
505
506        # Skip if points are collinear
507        if are_points_collinear(point_i, point1, point2):
508            continue
509
510        # Try different pairs of points until we find one that works
511        triangle_created = False
512        
513        # Try pairs in order of closeness
514        for i in range(len(distances) - 1):
515            # Skip if we've already created a triangle for this point
516            if triangle_created:
517                break
518                
519            for j in range(i + 1, len(distances)):
520                # Get the i-th and j-th closest points
521                closest1 = distances[i]
522                closest2 = distances[j]
523                
524                orig_idx1 = closest1[2]
525                orig_idx2 = closest2[2]
526                point1 = closest1[3]
527                point2 = closest2[3]
528                
529                # Skip if points are collinear
530                if are_points_collinear(point_i, point1, point2):
531                    continue
532                
533                # Check if any edge would cross existing triangles
534                edge_crosses = False
535                for p_idx, q_idx in [(orig_idx_i, orig_idx1), (orig_idx_i, orig_idx2), (orig_idx1, orig_idx2)]:
536                    if edge_crosses_triangles(p_idx, q_idx, triangles_geom, original_points):
537                        edge_crosses = True
538                        break
539                
540                if edge_crosses:
541                    continue  # Try next pair
542                
543                # Check if edges already exist
544                edge1_exists = edge_exists(orig_idx_i, orig_idx1)
545                edge2_exists = edge_exists(orig_idx_i, orig_idx2)
546                edge3_exists = edge_exists(orig_idx1, orig_idx2)
547                
548                # Sort vertices for consistent triangle representation
549                triangle_vertices = [orig_idx_i, orig_idx1, orig_idx2]
550                triangle_vertices.sort()
551                
552                # Create triangle with sorted vertices
553                triangle: Triangle_Indices = {
554                    'v1': triangle_vertices[0],
555                    'v2': triangle_vertices[1],
556                    'v3': triangle_vertices[2]
557                }
558                
559                # Add edges to adjacency matrix
560                if not edge1_exists:
561                    add_edge(orig_idx_i, orig_idx1)
562                
563                if not edge2_exists:
564                    add_edge(orig_idx_i, orig_idx2)
565                
566                if not edge3_exists:
567                    add_edge(orig_idx1, orig_idx2)
568                
569                # Mark all three vertices as used
570                used_in_triangle[orig_idx_i] = True
571                used_in_triangle[orig_idx1] = True
572                used_in_triangle[orig_idx2] = True
573                
574                # Add the triangle to result
575                triangles_geom.append(triangle)
576
577                triangle_created = True
578                break
579
580
581    # Create the Triangulation_Result dictionary with correct format
582    triangulation_result: Triangulation_Result = {
583        'points': original_points,
584        'triangles': triangles_geom
585    }
586    
587    print(f"Created {len(triangles_geom)} triangles")
588    return triangulation_result

592def triangulation_pipeline(pointSet: bytes) -> bytes :
593    """Réalise la triangulation d'un ensemble de points.
594
595    Args :
596          pointSet : L'ensemble des points dont on souhaite réaliser la triangulation,
597          au format binaire (PointSet)
598
599    Returns :
600            triangles : L'ensemble des triangles produits par la triangulation, 
601            au format binaire (Triangles)
602
603    Raises :
604        ----- 500 Internal Server Error ----------------------
605        Se produit quand l'algorithme de triangulation échoue, pour plusieurs raisons:
606        - La triangulation n'a pas convergé
607        - Le pointSet présente une incohérence de données
608        - etc...
609    """
610    # -----Verification--------
611    # Check PointSet integrity
612    validate_point_set(pointSet)
613    
614    # -----Pretraitement--------
615    # Convert to geometric format
616    pointSet_geom = decode_binary_point_set_to_geometric(pointSet)
617    
618    # -----Compute--------
619    # Algo the triangulation
620    triangulation_result = triangulation_compute(pointSet_geom)
621    
622    # -----Post-traitement--------------
623    # Convert to binary
624    triangulation_binary = encode_triangulation_result_to_binary(triangulation_result)
625    
626    # -----RETURN--------
627    # return triangulation as binary
628    return triangulation_binary

338def point_in_triangle(px, py, triangle, points):
339    """Check if a point (px, py) lies inside or on the boundary of a 2D triangle.
340
341    This function uses barycentric coordinates to determine if the test point
342    is within the triangular region defined by the vertices.
343
344    Args:
345        px (float): The X-coordinate of the point to check.
346        py (float): The Y-coordinate of the point to check.
347        triangle (dict): A dictionary representing the triangle, typically
348            containing vertex indices {'v1', 'v2', 'v3'}.
349        points (list[dict]): A list where each element is a dictionary
350            containing the coordinates of the vertices, e.g., [{'x': 1.0, 'y': 2.0}, ...].
351

354            triangle, False otherwise.
355

359    bx, by = points[v2]['x'], points[v2]['y']

367    gamma = 1.0 - alpha - beta
368

422def triangulation_compute(pointSet_geom: PointSet_Geom) -> Triangulation_Result:
423    """Réalise l'algorithme de triangulation pour un ensemble de points en entrée.
424
425    Implémente un algorithme glouton (greedy) et déterministe qui garantit
426    qu'un même ensemble de points produira toujours la même triangulation.
427    
428    L'algorithme procède de la manière suivante :
429    1. Pour chaque point, il identifie ses deux plus proches voisins
430    2. Il crée des triangles en connectant chaque point avec ses deux voisins
431    3. Il continue le processus tant qu'il reste des points non connectés
432    4. Il vérifie et évite les intersections entre triangles
433    
434    Note: Cet algorithme ne produit pas toutes les triangulations possibles,
435    mais une triangulation valide et cohérente.
436
437    Args:
438        pointSet_geom : L'ensemble des points dont on souhaite réaliser 
439          la triangulation, au format géométrique (PointSet_Geom).
440
441    Returns:
442        triangles_geom : L'ensemble des triangles produits par la triangulation,
443          au format géométrique (Triangles_Geom).
444
445    Raises:
446        - 500 Internal Server Error : L'algorithme de triangulation échoue
447
448    """
449    # Verify if enough points
450    if len(pointSet_geom) == 0:  # Empty
451        raise Exception("Le PointSet est vide")
452    elif len(pointSet_geom) < 3:  # Not enough point
453        raise Exception("Pas assez de points : le PointSet doit contenir au moins trois points")
454
455    # Check if all points are different
456    if not all_points_are_different(pointSet_geom):
457        raise Exception("Point Set non valide : au moins deux points sont identiques : tous doivent être différents")
458
459    # Check if all points are not collinear
460    if are_all_points_collinear(pointSet_geom):
461        raise Exception("Tous les points sont alignés.")
462
463    original_points = pointSet_geom.copy()
464    indexed_points = list(enumerate(original_points))
465    indexed_points.sort(key=lambda item: (item[1]['x'], item[1]['y']))
466
467    n = len(original_points)
468    triangles_geom: Triangles_Geom = []
469    
470    used_in_triangle = [False] * n
471    adjacency_matrix = [[False] * n for _ in range(n)]
472
473    def edge_exists(i, j):
474        return adjacency_matrix[i][j] or adjacency_matrix[j][i]
475
476    def add_edge(i, j):
477        adjacency_matrix[i][j] = True
478        adjacency_matrix[j][i] = True
479
480    # Process each point

482        orig_idx_i, point_i = indexed_points[current_idx]
483
484        if used_in_triangle[orig_idx_i]:
485            continue
486        
487        distances = []
488        for other_idx in range(n):
489            if other_idx == current_idx:
490                continue
491            
492            orig_idx_j, point_j = indexed_points[other_idx]
493            dist = two_points_distance(point_i, point_j)
494            distances.append((dist, other_idx, orig_idx_j, point_j))
495
496        distances.sort(key=lambda x: x[0])
497

499        closest1 = distances[0]
500        closest2 = distances[1]
501
502        orig_idx1 = closest1[2]
503        orig_idx2 = closest2[2]
504        point1 = closest1[3]
505        point2 = closest2[3]
506
507        # Skip if points are collinear
508        if are_points_collinear(point_i, point1, point2):
509            continue
510
511        # Try different pairs of points until we find one that works
512        triangle_created = False
513        
514        # Try pairs in order of closeness
515        for i in range(len(distances) - 1):
516            # Skip if we've already created a triangle for this point
517            if triangle_created:
518                break
519                
520            for j in range(i + 1, len(distances)):
521                # Get the i-th and j-th closest points
522                closest1 = distances[i]
523                closest2 = distances[j]
524                
525                orig_idx1 = closest1[2]

527                point1 = closest1[3]

535                edge_crosses = False

537                    if edge_crosses_triangles(p_idx, q_idx, triangles_geom, original_points):
538                        edge_crosses = True
539                        break
540                
541                if edge_crosses:
542                    continue  # Try next pair
543                
544                # Check if edges already exist
545                edge1_exists = edge_exists(orig_idx_i, orig_idx1)
546                edge2_exists = edge_exists(orig_idx_i, orig_idx2)
547                edge3_exists = edge_exists(orig_idx1, orig_idx2)
548                
549                # Sort vertices for consistent triangle representation
550                triangle_vertices = [orig_idx_i, orig_idx1, orig_idx2]
551                triangle_vertices.sort()
552                
553                # Create triangle with sorted vertices
554                triangle: Triangle_Indices = {
555                    'v1': triangle_vertices[0],
556                    'v2': triangle_vertices[1],
557                    'v3': triangle_vertices[2]
558                }
559                
560                # Add edges to adjacency matrix
561                if not edge1_exists:
562                    add_edge(orig_idx_i, orig_idx1)
563                
564                if not edge2_exists:
565                    add_edge(orig_idx_i, orig_idx2)

567                if not edge3_exists:

569                
570                # Mark all three vertices as used
571                used_in_triangle[orig_idx_i] = True
572                used_in_triangle[orig_idx1] = True
573                used_in_triangle[orig_idx2] = True
574                
575                # Add the triangle to result
576                triangles_geom.append(triangle)
577
578                triangle_created = True
579                break

583    triangulation_result: Triangulation_Result = {
584        'points': original_points,
585        'triangles': triangles_geom
586    }
587    
588    print(f"Created {len(triangles_geom)} triangles")
589    return triangulation_result

Réalise l'algorithme de triangulation pour un ensemble de points en entrée.

Implémente un algorithme glouton (greedy) et déterministe qui garantit qu'un même ensemble de points produira toujours la même triangulation.

L'algorithme procède de la manière suivante :

  1. Pour chaque point, il identifie ses deux plus proches voisins
  2. Il crée des triangles en connectant chaque point avec ses deux voisins
  3. Il continue le processus tant qu'il reste des points non connectés
  4. Il vérifie et évite les intersections entre triangles

Note: Cet algorithme ne produit pas toutes les triangulations possibles, mais une triangulation valide et cohérente.

Args: pointSet_geom : L'ensemble des points dont on souhaite réaliser la triangulation, au format géométrique (PointSet_Geom).

Returns: triangles_geom : L'ensemble des triangles produits par la triangulation, au format géométrique (Triangles_Geom).

Raises: - 500 Internal Server Error : L'algorithme de triangulation échoue
